def self.create(opts = {})
  sc = Class.new(self)
  sc.instance_eval do
    opts.each { |k,v| set k.to_sym, v }
    raise ArgumentError, "A session secret must be set with :session_secret" unless defined?(session_secret)
    add_cookie_middleware
    self
  end
end
